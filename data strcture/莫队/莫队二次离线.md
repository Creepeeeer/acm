是一种难的莫队题型，主要难在莫队窗口ql,qr滑动的时候不能O(1)的直接得到修改信息，如果使用普通莫队直接暴力滑窗是$O(nk\sqrt n)$的,如果利用二次离线莫队可以做到$O(nk+n\sqrt n)$

二次离线莫队通常适用于满足以下条件的题目：

- 可用莫队做。
- 莫队扩展或者删除一个点对答案的影响取决于当前区间的长度。
- 扩展或者删除一个点对答案的影响可用前缀写成差分的形式。

利用可差分信息，将答案差分化，分解成两个部分求出

执行普通莫队第一次离线得到第一部分差分量,并且生成第二部分离线查询任务(可用扫描线解决)，由于第二部分的划窗是由第一部分的普通莫队决定的，所以其实时间复杂度也是$O(n \sqrt n)$



[P4887 【模板】莫队二次离线（第十四分块(前体)） - 洛谷](https://www.luogu.com.cn/problem/P4887)

如果使用普通莫队的话每次滑窗的复杂度是$C_{14}^k$

设对于$a_{i}$区间[l,r]的贡献为$f(i,l,r)$

我们考虑滑窗[l,r]->[l,r+w]，产生的贡献为$\sum_{i=r+1}^{r+w} f(i,l,i-1)$

这个信息可差分为$f(i,1,i-1)-f(i,1,l-1)$

第一部分：

先预处理出二进制下恰好只有k个1的数karr[]

由于${a}\oplus{b}=c=>a \oplus c=b$

从左到右遍历序列 此时$f(i,1,i-1)$为$cnt[a[i]]$,遍历karr[]，让$(karr[x] \oplus a[i])$++

第二部分$f([r+1,r+w],1,i-1)$存储到vector node g[i-1]里，存储(r+1,r+w,符号，id),从1到n进行扫描线，这里的$[r+1,r+w]$其实就是普通莫队的滑窗距离(总和的时间复杂度为$O(n\sqrt n)$) ，扫描线的复杂度是$O(n C_{14}^{k})$

```c++
struct query{
    int l,r,id;
};
struct query2{
    int l,r,id,op;
};
vector<int>karr;
const int N=1e5,K=16384;
int a[N+1],cnt[K+1],bi[N+1];
ll pre[N+1],suf[N+2],an[N+1];
vector<query2>g[N+10],g2[N+10];
int blen;
int ca(int x){
    int an=0;
    while(x){
        an++;
        x&=(x-1);
    }
    return an;
}
signed main()
{
    ios::sync_with_stdio(false);
    cin.tie(0), cout.tie(0);
    int n,m,k;
    cin>>n>>m>>k;
    for(int i=1;i<=n;i++)cin>>a[i];
    vector<query>q(m);
    for(int i=0;i<m;i++){
        cin>>q[i].l>>q[i].r;
        q[i].id=i;
    }
    for(int i=0;i<K;i++){
        if(ca(i)==k)karr.push_back(i);
    }
    for(int i=1;i<=n;i++){
        pre[i]=pre[i-1]+cnt[a[i]];
        for(auto&x:karr){
            cnt[a[i]^x]++;
        }
    }
    memset(cnt,0,sizeof(cnt));
    for(int i=n;i>=1;i--){
        if(i!=n)suf[i]=suf[i+1]+cnt[a[i]];
        for(auto&x:karr){
            cnt[a[i]^x]++;
        }
    }
    memset(cnt,0,sizeof(cnt));
    blen=sqrt(n);
    for(int i=1;i<=n;i++)bi[i]=(i-1)/blen+1;
    sort(all(q),[](const query&x,const query&y){
        if(bi[x.l]!=bi[y.l])return bi[x.l]<bi[y.l];
        else return x.r<y.r;
    });
    int ql=1,qr=0;
    for(auto&[l,r,id]:q){
        if(qr<r){
            an[id]+=pre[r]-pre[qr];
            g[ql-1].push_back({qr+1,r,id,-1});
        }
        if(qr>r){
            an[id]-=pre[qr]-pre[r];
            g[ql-1].push_back({r+1,qr,id,1});
        }
        qr=r;
        if(ql>l){
            an[id]+=suf[l]-suf[ql];
            g2[qr+1].push_back({l,ql-1,id,-1});
        }
        if(ql<l){
            an[id]-=suf[ql]-suf[l];
            g2[qr+1].push_back({ql,l-1,id,1});
        }
        ql=l;
    }
    for(int i=0;i<=n;i++){
        if(i>0)for(auto&x:karr)cnt[a[i]^x]++;
        for(auto&[l,r,id,op]:g[i]){
            for(int j=l;j<=r;j++)an[id]+=1ll*op*cnt[a[j]];
        }
    }
    memset(cnt,0,sizeof(cnt));
    for(int i=n+1;i>=1;i--){
        if(i<=n)for(auto&x:karr)cnt[a[i]^x]++;
        for(auto&[l,r,id,op]:g2[i]){
            for(int j=l;j<=r;j++)an[id]+=1ll*op*cnt[a[j]];
        }
    }
    for(int i=1;i<m;i++)an[q[i].id]+=an[q[i-1].id];
    for(int i=0;i<m;i++)cout<<an[i]<<'\n';
    return 0;
}
```
