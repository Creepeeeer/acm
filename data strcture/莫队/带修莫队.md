比起普通莫队多了单点修改，此时按照修改来定时间戳，把第一次修改前的查询的时间戳都定为0，每次修改后把它后面的在下一个修改之前的查询的时间戳定为这次修改的的时间戳

**此时的排序策略**：

按左端点的块号升序，相等再按右端点的块号升序，相等再按时间戳升序

设此时块长为x，则块数为n/x

对于两个相邻的查询假如两个l,两个r都分别在相同的块里，l和r的改变量是x，时间戳t是单调的，此时复杂度为$O(x*n+n*({\frac{n}{x}})^2)$

如果不在同一个块还是同理的

此时把块长设置为$n^{\frac{2}{3}}$,时间复杂度是$O(n^{\frac{5}{3}})$

[P1903 [国家集训队\] 数颜色 / 维护队列 /【模板】带修莫队 - 洛谷](https://www.luogu.com.cn/problem/P1903)

```c++
struct query{
	int l,r,t,id;
};
const int N=1.4e5,M=1e6;
int a[N+1],bi[N+1],an[N+1];
int cnt[M+1];
pii change[N+1];
int qsize,usize,blen;
vector<query>q;
bool cmp(query&x,query&y){
	if(bi[x.l]!=bi[y.l])return bi[x.l]<bi[y.l];
	else if(bi[x.r]!=bi[y.r])return bi[x.r]<bi[y.r];
	else return x.t<y.t;
}
    int n,m;
    cin>>n>>m;
    for(int i=1;i<=n;i++)cin>>a[i];
    for(int i=0;i<m;i++){
    	char c;
    	cin>>c;
    	int l,r;
    	cin>>l>>r;
    	if(c=='Q'){
    		q.push_back({l,r,usize,qsize++});
    	}
    	else{
    		change[++usize]={l,r};
    	}
    }
    blen=pow(n,2.0/3);
    for(int i=1;i<=n;i++){
    	bi[i]=(i-1)/blen+1;
    }
    sort(all(q),cmp);
    int ql=1,qr=0,tt=0,sum=0;
    for(auto&[l,r,t,id]:q){
    	while(qr<r){
    		qr++;
    		if(++cnt[a[qr]]==1)sum++;
    	}
    	while(ql>l){
    		ql--;
    		if(++cnt[a[ql]]==1)sum++;
    	}
    	while(qr>r){
    		if(--cnt[a[qr]]==0)sum--;
    		qr--;
    	}
    	while(ql<l){
    		if(--cnt[a[ql]]==0)sum--;
    		ql++;
    	}
    	while(tt<t){
    		tt++;
    		auto&[p,v]=change[tt];
    		if(p>=ql&&p<=qr){
    		if(--cnt[a[p]]==0)sum--;
    		if(++cnt[v]==1)sum++;}
    		swap(v,a[p]);
    	}
    	while(tt>t){
    		auto&[p,v]=change[tt];
    		if(p>=ql&&p<=qr){
    		if(--cnt[a[p]]==0)sum--;
    		if(++cnt[v]==1)sum++;}
    		swap(v,a[p]);
    		tt--;
    	}
    	an[id]=sum;
    }
    for(int i=0;i<qsize;i++)cout<<an[i]<<'\n';
```
