# 高精度加

ps： 都是正数

```c++
string add(string s1, string s2) {
    if (s2.size() > s1.size())swap(s1, s2);
    int n = s1.size(), m = s2.size();
    vector<int>v1(n+5, 0), v2(n+5,0);
    for (int i = n - 1; i >= 0; i--) {
        v1[n - 1 - i] = s1[i]-'0';
    }
    for (int i = m - 1; i >= 0; i--) {
        v2[m - 1 - i] = s2[i]-'0';
    }
    for (int i = 0; i < m; i++) {
        v1[i] += v2[i];
        v1[i + 1] += v1[i] / 10;
        v1[i] %= 10;
    }
    int i = m;
    while (v1[i] >= 10) {
        v1[i + 1] += v1[i] / 10;
        v1[i] %= 10;
        i++;
    }
    if (v1[i] == 0)i--;
    int j = max(i, n - 1);
    string s;
    for (int i = j; i >= 0; i--) {
        s += v1[i] + '0';
    }
    return s;
}
```
