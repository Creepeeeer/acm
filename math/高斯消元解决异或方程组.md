void gaussxor(int n){
	for(int i=1;i<=n;i++){
		for(int j=1;j<=n;j++){
			if(j<i&&a[j][j])continue;
			if(a[j][i]){
				swap(a[i],a[j]);
				break;
			}
		}
		if(a[i][i]){
			for(int j=1;j<=n;j++){
				if(i!=j&&a[j][i]){
					for(int k=i;k<=n+1;k++){
						a[j][k]^=a[i][k];
					}
				}
			}
		}
	}
}
